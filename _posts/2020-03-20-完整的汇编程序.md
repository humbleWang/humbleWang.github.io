今天终于完成第一个完整的汇编程序了。因此将其记录在这里。

这个汇编题目出自王爽老师的《汇编语言》第三版 第八章的结尾。

简而言之，某公司有 1975-1995 年的发展情况。需要把其年份、收入、雇员数和计算后的每年平均工资放置在指定的数据段中。


以下附上完整的程序代码
```asm
assume cs:codesg, ds:data ,ss:stack

data segment

  db '1975','1976','1977','1978','1979','1980','1981','1982','1983' 
  db '1984','1985','1986','1987','1988','1989','1990','1991','1992' 
  db '1993','1994','1995' 
  ;以上表示 21 年的 21 个字符串

  dd 16,22,382,1356,2390,8000,16000,24486,50065,97479,140417,197514 
  dd 345980,590827,803530,1183000,1843000,2759000,3753000,4649000,5937000
  ;以上表示 21 年公司总收入的 21 个 dword 类型数据

  dw 3,7,9,13,28,38,130,220,476,778,1001,1442,2258,2793,4037,5635,8226 
  dw 11542,14430,15257,17800 
  ;以上表示 21 年公司雇员人数的 21 个 word 型数据

data ends

table segment

  db 21 dup ('year summ ne ?? ') 

table ends

stack segment
  db 21 dup ('                ')
stack ends

codesg segment

start: 

mov ax ,data
mov ds ,ax
mov ax ,table
mov es ,ax
mov ax ,stack
mov ss ,ax

mov sp ,0
mov cx ,21
mov si ,0

push_stack : 
  mov di ,21
  sub di ,cx
  push cx
  mov cx,4
  

  date :
  mov ax ,4
  sub ax ,cx
  mov dx ,ax
  add si ,ax

  mov bx ,di
  add bx ,bx
  add bx ,bx ; bx = 4 * di
  add bx ,ax ; 日期 = [ 4 * bi + ax]

  mov al , [bx] ;按顺序取到日期
   
  mov es:[si] ,al ;按顺序存日期

  sub si,dx
  loop date ;至此 已经完成日期的写入

  ;现在可用的寄存器为 ax、bx、dx、cx
  ;di 存储着循环的正次数 、si 存储写入的偏移位置
  ;读取数据 收入： [0054h + 2 * di、0056h + 2 * di]，
  ;雇员：[00A8h + 2 * di],
  ;收入写入 [si+5、si+7],雇员数 写入 [si + 10],平均收入写入 [si+13]

  add di ,di
  mov ax ,[0054h + di]
  mov dx ,[0056h + di]
  mov bx ,[00A8h + di]
  mov es:[si + 5] ,ax
  mov es:[si + 7] ,dx
  mov es:[si + 10] ,bx

  div bx
  mov es:[si + 13] ,ax

  pop cx
  add si, 16
loop push_stack

mov ax,4c00H
int 21H

codesg ends

end start

```
